{{- /* responsive-img.html — shared partial for responsive image output. SVGs: served as-is. Raster: resized to WebP with srcset. Callers pass "/images/foo.jpg" — leading "/" is stripped automatically. */ -}} {{- $src := .src -}} {{- $alt := .alt | default "" -}} {{- $widths := .widths | default (slice 400 800) -}} {{- $sizes := .sizes | default "100vw" -}} {{- $class := .class | default "" -}} {{- $loading := .loading | default "lazy" -}} {{- $fetchpriority := .fetchpriority | default "" -}} {{- $quality := .quality | default 82 -}} {{- if hasPrefix $src "/" -}} {{- $src = strings.TrimPrefix "/" $src -}} {{- end -}} {{- $resource := resources.Get $src -}} {{- if not $resource -}} {{- else if eq (path.Ext $resource.Name) ".svg" -}} {{- else -}} {{- $origWidth := $resource.Width -}} {{- $srcsetParts := slice -}} {{- $largestImg := $resource -}} {{- range $w := $widths -}} {{- if le $w $origWidth -}} {{- $resized := $resource.Resize (printf "%dx webp q%d" $w $quality) -}} {{- $srcsetParts = $srcsetParts | append (printf "%s %dw" $resized.RelPermalink $w) -}} {{- $largestImg = $resized -}} {{- end -}} {{- end -}} {{- if eq (len $srcsetParts) 0 -}} {{- $largestImg = $resource.Resize (printf "%dx webp q%d" $origWidth $quality) -}} {{- end -}} {{- end -}}
